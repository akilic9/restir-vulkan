78void VRE::VRE_glTFModel::LoadNode(glTFNode* parent, const tinygltf::Node& inputNode, const tinygltf::Model& model, ModelData& data)
79{
80 glTFNode* node = new glTFNode{};
81 node->mParent = parent;
82
83 // Node with children
84 if (inputNode.children.size() > 0)
85 for (size_t i = 0; i < inputNode.children.size(); i++)
86 LoadNode(node, model.nodes[inputNode.children[i]], model, data);
87
88 // If the node contains mesh data, we load vertices and indices from the buffers
89 // In glTF this is done via accessors and buffer views
90 if (inputNode.mesh > -1) {
91 const tinygltf::Mesh mesh = model.meshes[inputNode.mesh];
92
93 // Iterate through all primitives of this node's mesh
94 for (size_t i = 0; i < mesh.primitives.size(); i++) {
95 const tinygltf::Primitive& inputPrimitive = mesh.primitives[i];
96 uint32_t firstIndex = static_cast<uint32_t>(mIndexCount);
97 uint32_t vertexStart = static_cast<uint32_t>(mVertexCount);
98 uint32_t indexCount = 0;
99
100 // Vertices
101 {
102 const float* positionBuffer = nullptr;
103 const float* normalsBuffer = nullptr;
104 const float* texCoordsBuffer = nullptr;
105 const float* coloursBuffer = nullptr;
106 size_t vertexCount = 0;
107
108 // Get buffer data for vertex positions
109 if (inputPrimitive.attributes.find("POSITION") != inputPrimitive.attributes.end()) {
110 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("POSITION")->second];
111 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
112 positionBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
113 vertexCount = accessor.count;
114 }
115
116 // Get buffer data for vertex normals
117 if (inputPrimitive.attributes.find("NORMAL") != inputPrimitive.attributes.end()) {
118 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("NORMAL")->second];
119 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
120 normalsBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
121 }
122
123 // Get buffer data for vertex texture coordinates
124 // glTF supports multiple sets, we only load the first one
125 if (inputPrimitive.attributes.find("TEXCOORD_0") != inputPrimitive.attributes.end()) {
126 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("TEXCOORD_0")->second];
127 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
128 texCoordsBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
129 }
130
131 if (inputPrimitive.attributes.find("COLOR_0") != inputPrimitive.attributes.end()) {
132 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("COLOR_0")->second];
133 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
134 coloursBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
135 }
136
137 // Append data to model's vertex buffer
138 for (size_t v = 0; v < vertexCount; v++) {
139 Vertex vert{};
140 vert.mPosition = glm::vec4(glm::make_vec3(&positionBuffer[v * 3]), 1.0f);
141 vert.mNormal = glm::normalize(glm::vec3(normalsBuffer ? glm::make_vec3(&normalsBuffer[v * 3]) : glm::vec3(0.0f)));
142 vert.mTexCoord0 = texCoordsBuffer ? glm::make_vec2(&texCoordsBuffer[v * 2]) : glm::vec3(0.0f);
143 vert.mColor = coloursBuffer ? glm::make_vec3(&coloursBuffer[v * 3]) : glm::vec3(1.0f);
144 data.mVertices.push_back(vert);
145 mVertexCount++;
146 }
147 }
148 // Indices
149 {
150 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.indices];
151 const tinygltf::BufferView& bufferView = model.bufferViews[accessor.bufferView];
152 const tinygltf::Buffer& buffer = model.buffers[bufferView.buffer];
153
154 indexCount += static_cast<uint32_t>(accessor.count);
155
156 // glTF supports different component types of indices
157 switch (accessor.componentType) {
158 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_INT: {
159 const uint32_t* buf = reinterpret_cast<const uint32_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
160 for (size_t index = 0; index < accessor.count; index++) {
161 data.mIndices.push_back(buf[index] + vertexStart);
162 mIndexCount++;
163 }
164 break;
165 }
166 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
167 const uint16_t* buf = reinterpret_cast<const uint16_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
168 for (size_t index = 0; index < accessor.count; index++) {
169 data.mIndices.push_back(buf[index] + vertexStart);
170 mIndexCount++;
171 }
172 break;
173 }
174 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
175 const uint8_t* buf = reinterpret_cast<const uint8_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
176 for (size_t index = 0; index < accessor.count; index++) {
177 data.mIndices.push_back(buf[index] + vertexStart);
178 mIndexCount++;
179 }
180 break;
181 }
182 default:
183 std::cerr << "Index component type " << accessor.componentType << " not supported!" << std::endl;
184 return;
185 }
186 }
187 glTFPrimitive primitive{};
188 primitive.mFirstIndex = firstIndex;
189 primitive.mIndexCount = indexCount;
190 primitive.mMaterialIndex = inputPrimitive.material;
191 node->mMesh.mPrimitives.push_back(primitive);
192 }
193 }
194
195 if (parent)
196 parent->mChildren.push_back(node);
197 else
198 mNodes.push_back(node);
199}

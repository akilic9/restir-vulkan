27void VRE::VRE_glTFModel::LoadModel()
28{
29 auto startTime = std::chrono::high_resolution_clock::now();
30 tinygltf::Model model;
31 tinygltf::TinyGLTF gltfContext;
32 std::string error, warning;
33
34 std::string filePath = mFileFolder;
35
36 if (!gltfContext.LoadASCIIFromFile(&model, &error, &warning, filePath.append(mFileName).append(".gltf")))
37 throw std::runtime_error("Failed to load glTF file!");
38
39 LoadTextures(model);
40 LoadMaterials(model);
41
42 const tinygltf::Scene& scene = model.scenes[0];
43
44 ModelData data{};
45
46
47 for (size_t i = 0; i < scene.nodes.size(); i++) {
48 const tinygltf::Node node = model.nodes[scene.nodes[i]];
49 LoadNode(nullptr, node, model, data);
50 }
51
52 std::cout << "Model index count: " << mIndexCount << std::endl;
53 std::cout << "Model vertex count: " << mVertexCount << std::endl;
54 std::cout << "Model texture count: " << mTextures.size() << std::endl;
55
56 CreateVertexBuffers(data.mVertices);
57 CreateIndexBuffer(data.mIndices);
58
59 const auto currentTime = std::chrono::high_resolution_clock::now();
60 const float loadTime = std::chrono::duration<float, std::chrono::seconds::period>(currentTime - startTime).count();
61 std::cout << "Model load time: " << loadTime << std::endl;
62}

73void VRE::VRE_glTFModel::LoadMaterials(tinygltf::Model& model)
74{
75 mMaterials.resize(model.materials.size());
76 for (size_t i = 0; i < model.materials.size(); i++) {
77
78 tinygltf::Material inputMat = model.materials[i];
79
80 if (inputMat.values.find("baseColorFactor") != inputMat.values.end()) {
81 mMaterials[i].mBaseColorFactor = glm::make_vec4(inputMat.values["baseColorFactor"].ColorFactor().data());
82 }
83
84 if (inputMat.values.find("baseColorTexture") != inputMat.values.end()) {
85 mMaterials[i].mBaseColorTextureIndex = inputMat.values["baseColorTexture"].TextureIndex();
86 }
87 }
88
89 mTextureIndices.resize(model.textures.size());
90 for (size_t i = 0; i < model.textures.size(); i++) {
91 mTextureIndices[i] = model.textures[i].source;
92 }
93}

95void VRE::VRE_glTFModel::LoadNode(glTFNode* parent, const tinygltf::Node& inputNode, const tinygltf::Model& model, ModelData& data)
96{
97 glTFNode* node = new glTFNode{};
98 node->mParent = parent;
99
100 if (inputNode.children.size() > 0)
101 for (size_t i = 0; i < inputNode.children.size(); i++)
102 LoadNode(node, model.nodes[inputNode.children[i]], model, data);
103
104 if (inputNode.mesh > -1) {
105 const tinygltf::Mesh mesh = model.meshes[inputNode.mesh];
106
107 for (size_t i = 0; i < mesh.primitives.size(); i++) {
108 const tinygltf::Primitive& inputPrimitive = mesh.primitives[i];
109 uint32_t firstIndex = static_cast<uint32_t>(mIndexCount);
110 uint32_t vertexStart = static_cast<uint32_t>(mVertexCount);
111 uint32_t indexCount = 0;
112
113 const float* positionBuffer = nullptr;
114 const float* normalsBuffer = nullptr;
115 const float* texCoordsBuffer = nullptr;
116 const float* coloursBuffer = nullptr;
117 size_t vertexCount = 0;
118
119 if (inputPrimitive.attributes.find("POSITION") != inputPrimitive.attributes.end()) {
120 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("POSITION")->second];
121 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
122 positionBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
123 vertexCount = accessor.count;
124 }
125
126 if (inputPrimitive.attributes.find("NORMAL") != inputPrimitive.attributes.end()) {
127 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("NORMAL")->second];
128 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
129 normalsBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
130 }
131
132 if (inputPrimitive.attributes.find("TEXCOORD_0") != inputPrimitive.attributes.end()) {
133 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("TEXCOORD_0")->second];
134 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
135 texCoordsBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
136 }
137
138 if (inputPrimitive.attributes.find("COLOR_0") != inputPrimitive.attributes.end()) {
139 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.attributes.find("COLOR_0")->second];
140 const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
141 coloursBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[accessor.byteOffset + view.byteOffset]));
142 }
143
144 for (size_t v = 0; v < vertexCount; v++) {
145 Vertex vert{};
146 vert.mPosition = glm::vec4(glm::make_vec3(&positionBuffer[v * 3]), 1.0f);
147 vert.mNormal = glm::normalize(glm::vec3(normalsBuffer ? glm::make_vec3(&normalsBuffer[v * 3]) : glm::vec3(0.0f)));
148 vert.mTexCoord0 = texCoordsBuffer ? glm::make_vec2(&texCoordsBuffer[v * 2]) : glm::vec3(0.0f);
149 vert.mColor = coloursBuffer ? glm::make_vec3(&coloursBuffer[v * 3]) : glm::vec3(1.0f);
150 data.mVertices.push_back(vert);
151 mVertexCount++;
152 }
153
154 const tinygltf::Accessor& accessor = model.accessors[inputPrimitive.indices];
155 const tinygltf::BufferView& bufferView = model.bufferViews[accessor.bufferView];
156 const tinygltf::Buffer& buffer = model.buffers[bufferView.buffer];
157
158 indexCount += static_cast<uint32_t>(accessor.count);
159
160 // glTF supports different component types of indices
161 switch (accessor.componentType) {
162 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_INT: {
163 const uint32_t* buf = reinterpret_cast<const uint32_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
164 for (size_t index = 0; index < accessor.count; index++) {
165 data.mIndices.push_back(buf[index] + vertexStart);
166 mIndexCount++;
167 }
168 break;
169 }
170 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_SHORT: {
171 const uint16_t* buf = reinterpret_cast<const uint16_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
172 for (size_t index = 0; index < accessor.count; index++) {
173 data.mIndices.push_back(buf[index] + vertexStart);
174 mIndexCount++;
175 }
176 break;
177 }
178 case TINYGLTF_PARAMETER_TYPE_UNSIGNED_BYTE: {
179 const uint8_t* buf = reinterpret_cast<const uint8_t*>(&buffer.data[accessor.byteOffset + bufferView.byteOffset]);
180 for (size_t index = 0; index < accessor.count; index++) {
181 data.mIndices.push_back(buf[index] + vertexStart);
182 mIndexCount++;
183 }
184 break;
185 }
186 default:
187 std::cerr << "Index component type " << accessor.componentType << " not supported!" << std::endl;
188 return;
189 }
190
191 glTFPrimitive primitive{};
192 primitive.mFirstIndex = firstIndex;
193 primitive.mIndexCount = indexCount;
194 primitive.mMaterialIndex = inputPrimitive.material;
195 node->mMesh.mPrimitives.push_back(primitive);
196 }
197 }
198
199 if (parent)
200 parent->mChildren.push_back(node);
201 else
202 mNodes.push_back(node);
203}

220void VRE::VRE_glTFModel::RenderNode(VkCommandBuffer commandBuffer, glTFNode* node, VkPipelineLayout& pipelineLayout, VRE_DescriptorWriter& writer)
221{
222 if (node->mMesh.mPrimitives.size() > 0) {
223 for (glTFPrimitive& primitive : node->mMesh.mPrimitives) {
224 if (primitive.mIndexCount > 0) {
225 auto texture = mTextureIndices[mMaterials[primitive.mMaterialIndex].mBaseColorTextureIndex];
226 auto imageInfo = mTextures[texture]->GetDescImageInfo();
227 writer.WriteImage(1, &imageInfo);
228 VkDescriptorSet gameObjectDescriptorSet;
229 writer.Build(gameObjectDescriptorSet);
230 vkCmdBindDescriptorSets(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, pipelineLayout, 1, 1, &gameObjectDescriptorSet, 0, nullptr);
231 vkCmdDrawIndexed(commandBuffer, primitive.mIndexCount, 1, primitive.mFirstIndex, 0, 0);
232 }
233 }
234 }
235 for (auto& child : node->mChildren) {
236 RenderNode(commandBuffer, child, pipelineLayout, writer);
237 }
238}

240void VRE::VRE_glTFModel::CreateVertexBuffers(const std::vector<Vertex>& vertices)
241{
242 mVertexCount = static_cast<uint32_t>(vertices.size());
243
244 assert(mVertexCount >= 3 && "Vertex count must be at least 3!");
245 VkDeviceSize bufferSize = sizeof(vertices[0]) * mVertexCount;
246
247 uint32_t vertexSize = sizeof(vertices[0]);
248
249 VRE_Buffer stagingBuffer(mDevice, vertexSize, mVertexCount, VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
250 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT);
251
252 stagingBuffer.Map();
253 stagingBuffer.WriteToBuffer((void*)vertices.data());
254
255 mVertexBuffer = std::make_unique<VRE_Buffer>(mDevice, vertexSize, mVertexCount,
256 VK_BUFFER_USAGE_VERTEX_BUFFER_BIT | VK_BUFFER_USAGE_TRANSFER_DST_BIT,
257 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
258
259 mDevice.CopyBuffer(stagingBuffer.GetBuffer(), mVertexBuffer->GetBuffer(), bufferSize);
260}

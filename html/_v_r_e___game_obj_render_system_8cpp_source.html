26void VRE::VRE_GameObjRenderSystem::Render()
27{
28 mSharedContext->mObjectDescPools[mSharedContext->mRenderer->GetFrameIndex()]->ResetPool();
29
30 mPipeline->Bind(mSharedContext->mRenderer->GetCurrentCommandBuffer());
31
32 vkCmdBindDescriptorSets(mSharedContext->mRenderer->GetCurrentCommandBuffer(), VK_PIPELINE_BIND_POINT_GRAPHICS, mPipelineLayout, 0, 1,
33 &mSharedContext->mSceneDescriptorSets[mSharedContext->mRenderer->GetFrameIndex()], 0, nullptr);
34
35 for (auto& e : *mSharedContext->mGameObjMap) {
36 if (!e.second.mModel)
37 continue;
38
39 e.second.mModel->Bind(mSharedContext->mRenderer->GetCurrentCommandBuffer());
40
41 auto bufferInfo = e.second.GetBufferInfo();
42 auto writer = VRE_DescriptorWriter(*mDescSetLayout, *mSharedContext->mObjectDescPools[mSharedContext->mRenderer->GetFrameIndex()]);
43 writer.WriteBuffer(0, &bufferInfo);
44 e.second.mModel->Render(mSharedContext->mRenderer->GetCurrentCommandBuffer(), mPipelineLayout, writer);
45 }
46}

48void VRE::VRE_GameObjRenderSystem::CreatePipelineLayouts()
49{
50 VkPushConstantRange pushConstantRange{ VK_SHADER_STAGE_VERTEX_BIT | VK_SHADER_STAGE_FRAGMENT_BIT, 0, sizeof(GameObjectBufferData) };
51
52 mDescSetLayout = VRE_DescriptorSetLayout::Builder(*mSharedContext->mDevice)
53 .AddBinding(0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_VERTEX_BIT | VK_SHADER_STAGE_FRAGMENT_BIT)
54 .AddBinding(1, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, VK_SHADER_STAGE_FRAGMENT_BIT)
55 .Build();
56
57 std::vector<VkDescriptorSetLayout> descSetLayouts{ mSharedContext->mGlobalDescSetLayout->GetDescriptorSetLayout(),
58 mDescSetLayout->GetDescriptorSetLayout()};
59
60 VkPipelineLayoutCreateInfo pipelineLayoutInfo{};
61 pipelineLayoutInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_LAYOUT_CREATE_INFO;
62 pipelineLayoutInfo.setLayoutCount = static_cast<uint32_t>(descSetLayouts.size());
63 pipelineLayoutInfo.pSetLayouts = descSetLayouts.data();
64 pipelineLayoutInfo.pushConstantRangeCount = 1;
65 pipelineLayoutInfo.pPushConstantRanges = &pushConstantRange;
66
67 if (vkCreatePipelineLayout(mSharedContext->mDevice->GetVkDevice(), &pipelineLayoutInfo, nullptr, &mPipelineLayout) != VK_SUCCESS)
68 throw std::runtime_error("Failed to create pipeline layout!");
69}

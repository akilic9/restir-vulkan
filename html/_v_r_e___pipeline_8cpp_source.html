26std::vector<char> VRE::VRE_Pipeline::ReadFile(const std::string& filePath)
27{
28 std::ifstream file{ filePath, std::ios::ate | std::ios::binary };
29
30 if (!file.is_open())
31 std::cout << "Could not open file:" << filePath << std::endl;
32
33 size_t fileSize = static_cast<size_t>(file.tellg());
34 std::vector<char> buffer(fileSize);
35
36 file.seekg(0);
37 file.read(buffer.data(), fileSize);
38
39 file.close();
40 return buffer;
41}

43void VRE::VRE_Pipeline::CreateGraphicsPipeline(const PipelineConfigInfo& configInfo, const std::string& vertShaderPath, const std::string& fragShaderPath)
44{
45 auto vertCode = ReadFile(vertShaderPath);
46 auto fragCode = ReadFile(fragShaderPath);
47
48 CreateShaderModule(vertCode, &mVertShaderModule);
49 CreateShaderModule(fragCode, &mFragShaderModule);
50
51 VkPipelineShaderStageCreateInfo shaderStages[2];
52 shaderStages[0].sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
53 shaderStages[0].stage = VK_SHADER_STAGE_VERTEX_BIT;
54 shaderStages[0].module = mVertShaderModule;
55 shaderStages[0].pName = "main";
56 shaderStages[0].flags = 0;
57 shaderStages[0].pNext = nullptr;
58 shaderStages[0].pSpecializationInfo = nullptr;
59 shaderStages[1].sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
60 shaderStages[1].stage = VK_SHADER_STAGE_FRAGMENT_BIT;
61 shaderStages[1].module = mFragShaderModule;
62 shaderStages[1].pName = "main";
63 shaderStages[1].flags = 0;
64 shaderStages[1].pNext = nullptr;
65 shaderStages[1].pSpecializationInfo = nullptr;
66
67 auto& bindingDescriptions = configInfo.mBindingDescriptions;
68 auto& attributeDescriptions = configInfo.mAttributeDescriptions;
69 VkPipelineVertexInputStateCreateInfo vertexInputInfo{};
70 vertexInputInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
71 vertexInputInfo.vertexAttributeDescriptionCount =
72 static_cast<uint32_t>(attributeDescriptions.size());
73 vertexInputInfo.vertexBindingDescriptionCount = static_cast<uint32_t>(bindingDescriptions.size());
74 vertexInputInfo.pVertexAttributeDescriptions = attributeDescriptions.data();
75 vertexInputInfo.pVertexBindingDescriptions = bindingDescriptions.data();
76
77 VkGraphicsPipelineCreateInfo pipelineInfo{};
78 pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
79 pipelineInfo.stageCount = 2;
80 pipelineInfo.pStages = shaderStages;
81 pipelineInfo.pVertexInputState = &vertexInputInfo;
82 pipelineInfo.pInputAssemblyState = &configInfo.mInputAssemblyInfo;
83 pipelineInfo.pViewportState = &configInfo.mViewportInfo;
84 pipelineInfo.pRasterizationState = &configInfo.mRasterizationInfo;
85 pipelineInfo.pMultisampleState = &configInfo.mMultisampleInfo;
86 pipelineInfo.pColorBlendState = &configInfo.mColorBlendInfo;
87 pipelineInfo.pDepthStencilState = &configInfo.mDepthStencilInfo;
88 pipelineInfo.pDynamicState = &configInfo.mDynamicStateInfo;
89 pipelineInfo.layout = configInfo.mPipelineLayout;
90 pipelineInfo.renderPass = configInfo.mRenderPass;
91 pipelineInfo.subpass = configInfo.mSubpass;
92
93 pipelineInfo.basePipelineIndex = -1;
94 pipelineInfo.basePipelineHandle = VK_NULL_HANDLE;
95
96 if (vkCreateGraphicsPipelines(mDevice.GetVkDevice(), VK_NULL_HANDLE, 1, &pipelineInfo, nullptr, &mGraphicsPipeline) != VK_SUCCESS)
97 std::cout << "Failed to create graphics pipeline!" << std::endl;
98}

100void VRE::VRE_Pipeline::CreateShaderModule(const std::vector<char>& shaderCode, VkShaderModule* shaderModule)
101{
102 VkShaderModuleCreateInfo createInfo{};
103 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
104 createInfo.codeSize = shaderCode.size();
105 createInfo.pCode = reinterpret_cast<const uint32_t*>(shaderCode.data());
106
107 if (vkCreateShaderModule(mDevice.GetVkDevice(), &createInfo, nullptr, shaderModule) != VK_SUCCESS)
108 std::cout << "Failed to create shader module!" << std::endl;
109}

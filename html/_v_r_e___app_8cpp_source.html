43void VRE::VRE_App::Run()
44{
45 auto startTime = std::chrono::high_resolution_clock::now();
46
47 while (!mWindow.ShouldClose()) {
48 glfwPollEvents();
49
50 const auto currentTime = std::chrono::high_resolution_clock::now();
51 const float deltaTime = std::chrono::duration<float, std::chrono::seconds::period>(currentTime - startTime).count();
52 //std::cout << 1.0f / deltaTime << std::endl;
53 startTime = currentTime;
54
55 float aspRatio = mRenderer.GetAspectRatio();
56 mCamera.SetPerspectiveProjection(glm::radians(50.f), aspRatio, 0.1f, 1000.f);
57
58 // TODO: Can probably optimize this using GLFW callback function.
59 mInputListener.Move(mWindow.GetGLFWwindow(), deltaTime, mCamera);
60
61 if (auto commandBuffer = mRenderer.BeginDraw()) {
62 const int frameIndex = mRenderer.GetFrameIndex();
63 UBO ubo;
64
65 Update(deltaTime, ubo);
66 Render(ubo);
67
68 mRenderer.EndDraw();
69 }
70 }
71 vkDeviceWaitIdle(mDevice.GetVkDevice());
72}

74void VRE::VRE_App::Init()
75{
76 //Create descriptor pool for global data.
77 mDescriptorPool = VRE_DescriptorPool::Builder(mDevice)
78 .SetMaxSets(VRE_SwapChain::MAX_FRAMES_IN_FLIGHT)
79 .AddPoolSize(VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VRE_SwapChain::MAX_FRAMES_IN_FLIGHT)
80 .Build();
81
82 mSceneContext.mObjectDescPools.resize(VRE_SwapChain::MAX_FRAMES_IN_FLIGHT);
83 auto objPoolBuilder = VRE_DescriptorPool::Builder(mDevice)
84 .SetMaxSets(1000)
85 .AddPoolSize(VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 1000)
86 .AddPoolSize(VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, 1000)
87 .SetPoolFlags(VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT);
88
89 for (int i = 0; i < mSceneContext.mObjectDescPools.size(); i++) {
90 mSceneContext.mObjectDescPools[i] = objPoolBuilder.Build();
91 }
92
93 for (int i = 0; i < VRE_SwapChain::MAX_FRAMES_IN_FLIGHT; i++) {
94 mSceneUBOs.push_back(std::make_unique<VRE_Buffer>(mDevice, sizeof(UBO), 1, VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT, VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT));
95 mSceneUBOs[i]->Map();
96 }
97
98 mSceneContext.mGlobalDescSetLayout = VRE_DescriptorSetLayout::Builder(mDevice)
99 .AddBinding(0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_ALL_GRAPHICS)
100 .Build();
101
102
103 for (int i = 0; i < VRE_SwapChain::MAX_FRAMES_IN_FLIGHT; i++) {
104 auto bufferInfo = mSceneUBOs[i]->DescriptorInfo();
105 mSceneContext.mSceneDescriptorSets.push_back(VkDescriptorSet());
106 VRE_DescriptorWriter(*mSceneContext.mGlobalDescSetLayout, *mDescriptorPool)
107 .WriteBuffer(0, &bufferInfo)
108 .Build(mSceneContext.mSceneDescriptorSets[i]);
109 }
110
111 LoadObjects();
112 mSceneContext.mGameObjMap = &mGameObjectManager.GetGameObjectsMap();
113 mPLRenderSystem.Init();
114 mGameObjectManager.Init();
115 mGameObjRenderSystem.Init();
116 mCamera = VRE_Camera();
117 mInputListener = VRE_InputListener();
118}

120void VRE::VRE_App::Update(float dt, UBO& ubo)
121{
122 ubo.mProjectionMat = mCamera.GetProjection();
123 ubo.mViewMat = mCamera.GetViewMat();
124 ubo.mInvViewMat = mCamera.GetInvViewMat();
125 mPLRenderSystem.Update(ubo, dt);
126
127 mSceneUBOs[mRenderer.GetFrameIndex()]->WriteToBuffer(&ubo);
128 mSceneUBOs[mRenderer.GetFrameIndex()]->Flush();
129
130 mGameObjectManager.Update(dt);
131}

31void VRE::VRE_App::Run()
32{
33 auto totalFrames = 0.f;
34 int frameCount = 0;
35
36 auto startTime = std::chrono::high_resolution_clock::now();
37
38 while (!mWindow.ShouldClose()) {
39 glfwPollEvents();
40
41 const auto currentTime = std::chrono::high_resolution_clock::now();
42 const float deltaTime = std::chrono::duration<float, std::chrono::seconds::period>(currentTime - startTime).count();
43 //std::cout << 1.0f / deltaTime << std::endl;
44 totalFrames += (1.0f / deltaTime);
45 frameCount++;
46 startTime = currentTime;
47
48 float aspRatio = mRenderer.GetAspectRatio();
49 mCamera.SetPerspectiveProjection(glm::radians(50.f), aspRatio, 0.1f, 1000.f);
50
51 // TODO: Can probably optimize this using GLFW callback function.
52 mInputListener.Move(mWindow.GetGLFWwindow(), deltaTime, mCamera);
53
54 if (auto commandBuffer = mRenderer.BeginDraw()) {
55 const int frameIndex = mRenderer.GetFrameIndex();
56 UBO ubo;
57
58 Update(deltaTime, ubo);
59 Render(ubo);
60
61 mRenderer.EndDraw();
62 }
63 }
64
65 std::cout << "Avg. frame rate for run: " << totalFrames / frameCount << std::endl;
66 vkDeviceWaitIdle(mDevice.GetVkDevice());
67}

69void VRE::VRE_App::Init()
70{
71 //Create descriptor pool for global data.
72 mDescriptorPool = VRE_DescriptorPool::Builder(mDevice)
73 .SetMaxSets(VRE_SwapChain::MAX_FRAMES_IN_FLIGHT)
74 .AddPoolSize(VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VRE_SwapChain::MAX_FRAMES_IN_FLIGHT)
75 .Build();
76
77 mSceneContext.mObjectDescPools.resize(VRE_SwapChain::MAX_FRAMES_IN_FLIGHT);
78 auto objPoolBuilder = VRE_DescriptorPool::Builder(mDevice)
79 .SetMaxSets(1000)
80 .AddPoolSize(VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 1000)
81 .AddPoolSize(VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, 1000)
82 .SetPoolFlags(VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT);
83
84 for (int i = 0; i < mSceneContext.mObjectDescPools.size(); i++) {
85 mSceneContext.mObjectDescPools[i] = objPoolBuilder.Build();
86 }
87
88 for (int i = 0; i < VRE_SwapChain::MAX_FRAMES_IN_FLIGHT; i++) {
89 mSceneUBOs.push_back(std::make_unique<VRE_Buffer>(mDevice, sizeof(UBO), 1, VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT, VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT));
90 mSceneUBOs[i]->Map();
91 }
92
93 mSceneContext.mGlobalDescSetLayout = VRE_DescriptorSetLayout::Builder(mDevice)
94 .AddBinding(0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_ALL_GRAPHICS)
95 .Build();
96
97
98 for (int i = 0; i < VRE_SwapChain::MAX_FRAMES_IN_FLIGHT; i++) {
99 auto bufferInfo = mSceneUBOs[i]->DescriptorInfo();
100 mSceneContext.mSceneDescriptorSets.push_back(VkDescriptorSet());
101 VRE_DescriptorWriter(*mSceneContext.mGlobalDescSetLayout, *mDescriptorPool)
102 .WriteBuffer(0, &bufferInfo)
103 .Build(mSceneContext.mSceneDescriptorSets[i]);
104 }
105
106 LoadObjects();
107 mSceneContext.mGameObjMap = &mGameObjectManager.GetGameObjectsMap();
108 mPLRenderSystem.Init();
109 mGameObjectManager.Init();
110 mGameObjRenderSystem.Init();
111 mCamera = VRE_Camera();
112 mInputListener = VRE_InputListener();
113}

26VkCommandBuffer VRE::VRE_Renderer::BeginDraw()
27{
28 assert(!mDrawStarted && "Can't call BeginDraw while already in progress.");
29
30 auto result = mSwapChain->AcquireNextImage(&mCurrentImageIndex);
31
32 if (result == VK_ERROR_OUT_OF_DATE_KHR) {
33 RecreateSwapChain();
34 return nullptr;
35 }
36
37 if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR)
38 throw std::runtime_error("Failed to acquire swap chain image!");
39
40 mDrawStarted = true;
41 auto commandBuffer = GetCurrentCommandBuffer();
42
43 VkCommandBufferBeginInfo beginInfo{};
44 beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
45
46 if (vkBeginCommandBuffer(commandBuffer, &beginInfo) != VK_SUCCESS)
47 throw std::runtime_error("Failed to begin recording command buffer!");
48
49 return commandBuffer;
50}

52void VRE::VRE_Renderer::EndDraw()
53{
54 assert(mDrawStarted && "Can't call EndDraw while frame is not in progress.");
55 auto commandBuffer = GetCurrentCommandBuffer();
56
57 if (vkEndCommandBuffer(commandBuffer) != VK_SUCCESS)
58 throw std::runtime_error("Failed to record command buffer!");
59
60 auto result = mSwapChain->SubmitCommandBuffers(&commandBuffer, &mCurrentImageIndex);
61 if (result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR ||
62 mWindow.HasWindowResized()) {
63 mWindow.ResetWindowResizedFlag();
64 RecreateSwapChain();
65 }
66 else if (result != VK_SUCCESS)
67 throw std::runtime_error("Failed to present swap chain image!");
68
69 mDrawStarted = false;
70 mCurrentFrameIndex = (mCurrentFrameIndex + 1) % VRE_SwapChain::MAX_FRAMES_IN_FLIGHT;
71}

73void VRE::VRE_Renderer::BeginSwapChainRenderPass(VkCommandBuffer commandBuffer)
74{
75 assert(mDrawStarted && "Can't call BeginSwapChainRenderPass if frame is not in progress.");
76 assert(commandBuffer == GetCurrentCommandBuffer() && "Can't begin render pass on command buffer from a different frame.");
77
78 VkRenderPassBeginInfo renderPassInfo{};
79 renderPassInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
80 renderPassInfo.renderPass = mSwapChain->GetRenderPass();
81 renderPassInfo.framebuffer = mSwapChain->GetFrameBuffer(mCurrentImageIndex);
82
83 renderPassInfo.renderArea.offset = { 0, 0 };
84 renderPassInfo.renderArea.extent = mSwapChain->GetSwapChainExtent();
85
86 std::array<VkClearValue, 2> clearValues{};
87 clearValues[0].color = { 0.01f, 0.01f, 0.01f, 1.0f };
88 clearValues[1].depthStencil = { 1.0f, 0 };
89 renderPassInfo.clearValueCount = static_cast<uint32_t>(clearValues.size());
90 renderPassInfo.pClearValues = clearValues.data();
91
92 vkCmdBeginRenderPass(commandBuffer, &renderPassInfo, VK_SUBPASS_CONTENTS_INLINE);
93
94 VkViewport viewport{};
95 viewport.x = 0.0f;
96 viewport.y = 0.0f;
97 viewport.width = static_cast<float>(mSwapChain->GetSwapChainExtent().width);
98 viewport.height = static_cast<float>(mSwapChain->GetSwapChainExtent().height);
99 viewport.minDepth = 0.0f;
100 viewport.maxDepth = 1.0f;
101 VkRect2D scissor{ {0, 0}, mSwapChain->GetSwapChainExtent() };
102 vkCmdSetViewport(commandBuffer, 0, 1, &viewport);
103 vkCmdSetScissor(commandBuffer, 0, 1, &scissor);
104}

128void VRE::VRE_Renderer::RecreateSwapChain()
129{
130 auto extent = mWindow.GetExtent();
131
132 while (extent.width == 0 || extent.height == 0) { //Pause if the window is minimized as it can cause swapchain to become out of date.
133 extent = mWindow.GetExtent();
134 glfwWaitEvents();
135 }
136
137 vkDeviceWaitIdle(mDevice.GetVkDevice());
138
139 if (mSwapChain == nullptr)
140 mSwapChain = std::make_unique<VRE_SwapChain>(mDevice, extent);
141 else {
142 std::shared_ptr<VRE_SwapChain> oldChain = std::move(mSwapChain);
143 mSwapChain = std::make_unique<VRE_SwapChain>(mDevice, extent, oldChain);
144
145 if (!oldChain->CompareSwapFormats(*mSwapChain.get()))
146 throw std::runtime_error("Swap chain image (or depth) format has changed!");
147 }
148}

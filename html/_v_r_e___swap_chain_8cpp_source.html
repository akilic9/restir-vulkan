34 VRE_SwapChain::~VRE_SwapChain()
35 {
36 for (auto imageView : mSwapChainImageViews)
37 vkDestroyImageView(mDevice.GetVkDevice(), imageView, nullptr);
38
39 mSwapChainImageViews.clear();
40
41 if (mSwapChain != nullptr) {
42 vkDestroySwapchainKHR(mDevice.GetVkDevice(), mSwapChain, nullptr);
43 mSwapChain = nullptr;
44 }
45
46 for (int i = 0; i < mDepthImages.size(); i++) {
47 vkDestroyImageView(mDevice.GetVkDevice(), mDepthImageViews[i], nullptr);
48 vkDestroyImage(mDevice.GetVkDevice(), mDepthImages[i], nullptr);
49 vkFreeMemory(mDevice.GetVkDevice(), mDepthImageMemorys[i], nullptr);
50 }
51
52 for (auto framebuffer : mSwapChainFramebuffers)
53 vkDestroyFramebuffer(mDevice.GetVkDevice(), framebuffer, nullptr);
54
55 vkDestroyRenderPass(mDevice.GetVkDevice(), mRenderPass, nullptr);
56
57 // cleanup synchronization objects
58 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
59 vkDestroySemaphore(mDevice.GetVkDevice(), mRenderFinishedSemaphores[i], nullptr);
60 vkDestroySemaphore(mDevice.GetVkDevice(), mImageAvailableSemaphores[i], nullptr);
61 vkDestroyFence(mDevice.GetVkDevice(), mInFlightFences[i], nullptr);
62 }
63 }

75 VkResult VRE_SwapChain::AcquireNextImage(uint32_t *imageIndex)
76 {
77 vkWaitForFences(mDevice.GetVkDevice(), 1, &mInFlightFences[mCurrentFrame], VK_TRUE, std::numeric_limits<uint64_t>::max());
78
79 VkResult result = vkAcquireNextImageKHR(mDevice.GetVkDevice(), mSwapChain, std::numeric_limits<uint64_t>::max(),
80 mImageAvailableSemaphores[mCurrentFrame], // must be a not signaled semaphore
81 VK_NULL_HANDLE, imageIndex);
82
83 return result;
84 }

86 VkResult VRE_SwapChain::SubmitCommandBuffers(const VkCommandBuffer *buffers, uint32_t *imageIndex)
87 {
88 if (mImagesInFlight[*imageIndex] != VK_NULL_HANDLE)
89 vkWaitForFences(mDevice.GetVkDevice(), 1, &mImagesInFlight[*imageIndex], VK_TRUE, UINT64_MAX);
90
91 mImagesInFlight[*imageIndex] = mInFlightFences[mCurrentFrame];
92
93 VkSubmitInfo submitInfo = {};
94 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
95
96 VkSemaphore waitSemaphores[] = {mImageAvailableSemaphores[mCurrentFrame]};
97 VkPipelineStageFlags waitStages[] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
98 submitInfo.waitSemaphoreCount = 1;
99 submitInfo.pWaitSemaphores = waitSemaphores;
100 submitInfo.pWaitDstStageMask = waitStages;
101
102 submitInfo.commandBufferCount = 1;
103 submitInfo.pCommandBuffers = buffers;
104
105 VkSemaphore signalSemaphores[] = {mRenderFinishedSemaphores[mCurrentFrame]};
106 submitInfo.signalSemaphoreCount = 1;
107 submitInfo.pSignalSemaphores = signalSemaphores;
108
109 vkResetFences(mDevice.GetVkDevice(), 1, &mInFlightFences[mCurrentFrame]);
110
111 if (vkQueueSubmit(mDevice.GraphicsQueue(), 1, &submitInfo, mInFlightFences[mCurrentFrame]) != VK_SUCCESS)
112 throw std::runtime_error("Failed to submit draw command buffer!");
113
114 VkPresentInfoKHR presentInfo = {};
115 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
116
117 presentInfo.waitSemaphoreCount = 1;
118 presentInfo.pWaitSemaphores = signalSemaphores;
119
120 VkSwapchainKHR swapChains[] = {mSwapChain};
121 presentInfo.swapchainCount = 1;
122 presentInfo.pSwapchains = swapChains;
123
124 presentInfo.pImageIndices = imageIndex;
125
126 auto result = vkQueuePresentKHR(mDevice.PresentQueue(), &presentInfo);
127
128 mCurrentFrame = (mCurrentFrame + 1) % MAX_FRAMES_IN_FLIGHT;
129
130 return result;
131 }

133 void VRE_SwapChain::CreateSwapChain() {
134 SwapChainSupportDetails swapChainSupport = mDevice.GetSwapChainSupport();
135
136 VkSurfaceFormatKHR surfaceFormat = ChooseSwapSurfaceFormat(swapChainSupport.mFormats);
137 VkPresentModeKHR presentMode = ChooseSwapPresentMode(swapChainSupport.mPresentModes);
138 VkExtent2D extent = ChooseSwapExtent(swapChainSupport.mCapabilities);
139
140 uint32_t imageCount = swapChainSupport.mCapabilities.minImageCount + 1;
141
142 if (swapChainSupport.mCapabilities.maxImageCount > 0
143 && imageCount > swapChainSupport.mCapabilities.maxImageCount) {
144 imageCount = swapChainSupport.mCapabilities.maxImageCount;
145 }
146
147 VkSwapchainCreateInfoKHR createInfo = {};
148 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
149 createInfo.surface = mDevice.Surface();
150
151 createInfo.minImageCount = imageCount;
152 createInfo.imageFormat = surfaceFormat.format;
153 createInfo.imageColorSpace = surfaceFormat.colorSpace;
154 createInfo.imageExtent = extent;
155 createInfo.imageArrayLayers = 1;
156 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
157
158 QueueFamilyIndices indices = mDevice.FindPhysicalQueueFamilies();
159 uint32_t queueFamilyIndices[] = {indices.mGraphicsFamily, indices.mPresentFamily};
160
161 if (indices.mGraphicsFamily != indices.mPresentFamily) {
162 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
163 createInfo.queueFamilyIndexCount = 2;
164 createInfo.pQueueFamilyIndices = queueFamilyIndices;
165 }
166 else {
167 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
168 createInfo.queueFamilyIndexCount = 0;
169 createInfo.pQueueFamilyIndices = nullptr;
170 }
171
172 createInfo.preTransform = swapChainSupport.mCapabilities.currentTransform;
173 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
174
175 createInfo.presentMode = presentMode;
176 createInfo.clipped = VK_TRUE;
177
178 createInfo.oldSwapchain = mOldSwapChain == nullptr ? VK_NULL_HANDLE : mOldSwapChain->mSwapChain;
179
180 if (vkCreateSwapchainKHR(mDevice.GetVkDevice(), &createInfo, nullptr, &mSwapChain) != VK_SUCCESS)
181 throw std::runtime_error("Failed to create swap chain!");
182
183 vkGetSwapchainImagesKHR(mDevice.GetVkDevice(), mSwapChain, &imageCount, nullptr);
184 mSwapChainImages.resize(imageCount);
185 vkGetSwapchainImagesKHR(mDevice.GetVkDevice(), mSwapChain, &imageCount, mSwapChainImages.data());
186
187 mSwapChainImageFormat = surfaceFormat.format;
188 mSwapChainExtent = extent;
189 }

212 void VRE_SwapChain::CreateRenderPass()
213 {
214 VkAttachmentDescription depthAttachment{};
215 depthAttachment.format = FindDepthFormat();
216 depthAttachment.samples = VK_SAMPLE_COUNT_1_BIT;
217 depthAttachment.loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
218 depthAttachment.storeOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
219 depthAttachment.stencilLoadOp = VK_ATTACHMENT_LOAD_OP_DONT_CARE;
220 depthAttachment.stencilStoreOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
221 depthAttachment.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
222 depthAttachment.finalLayout = VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
223
224 VkAttachmentReference depthAttachmentRef{};
225 depthAttachmentRef.attachment = 1;
226 depthAttachmentRef.layout = VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
227
228 VkAttachmentDescription colorAttachment = {};
229 colorAttachment.format = GetSwapChainImageFormat();
230 colorAttachment.samples = VK_SAMPLE_COUNT_1_BIT;
231 colorAttachment.loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
232 colorAttachment.storeOp = VK_ATTACHMENT_STORE_OP_STORE;
233 colorAttachment.stencilStoreOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
234 colorAttachment.stencilLoadOp = VK_ATTACHMENT_LOAD_OP_DONT_CARE;
235 colorAttachment.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
236 colorAttachment.finalLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
237
238 VkAttachmentReference colorAttachmentRef = {};
239 colorAttachmentRef.attachment = 0;
240 colorAttachmentRef.layout = VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL;
241
242 VkSubpassDescription subpass = {};
243 subpass.pipelineBindPoint = VK_PIPELINE_BIND_POINT_GRAPHICS;
244 subpass.colorAttachmentCount = 1;
245 subpass.pColorAttachments = &colorAttachmentRef;
246 subpass.pDepthStencilAttachment = &depthAttachmentRef;
247
248 VkSubpassDependency dependency = {};
249 dependency.srcSubpass = VK_SUBPASS_EXTERNAL;
250 dependency.srcAccessMask = 0;
251 dependency.srcStageMask = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT | VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT;
252 dependency.dstSubpass = 0;
253 dependency.dstStageMask = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT | VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT;
254 dependency.dstAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
255
256 std::array<VkAttachmentDescription, 2> attachments = {colorAttachment, depthAttachment};
257 VkRenderPassCreateInfo renderPassInfo = {};
258 renderPassInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_CREATE_INFO;
259 renderPassInfo.attachmentCount = static_cast<uint32_t>(attachments.size());
260 renderPassInfo.pAttachments = attachments.data();
261 renderPassInfo.subpassCount = 1;
262 renderPassInfo.pSubpasses = &subpass;
263 renderPassInfo.dependencyCount = 1;
264 renderPassInfo.pDependencies = &dependency;
265
266 if (vkCreateRenderPass(mDevice.GetVkDevice(), &renderPassInfo, nullptr, &mRenderPass) != VK_SUCCESS)
267 throw std::runtime_error("Failed to create render pass!");
268 }

336 void VRE_SwapChain::CreateSyncObjects()
337 {
338 mImageAvailableSemaphores.resize(MAX_FRAMES_IN_FLIGHT);
339 mRenderFinishedSemaphores.resize(MAX_FRAMES_IN_FLIGHT);
340 mInFlightFences.resize(MAX_FRAMES_IN_FLIGHT);
341 mImagesInFlight.resize(GetImageCount(), VK_NULL_HANDLE);
342
343 VkSemaphoreCreateInfo semaphoreInfo = {};
344 semaphoreInfo.sType = VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO;
345
346 VkFenceCreateInfo fenceInfo = {};
347 fenceInfo.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
348 fenceInfo.flags = VK_FENCE_CREATE_SIGNALED_BIT;
349
350 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
351 if (vkCreateSemaphore(mDevice.GetVkDevice(), &semaphoreInfo, nullptr, &mImageAvailableSemaphores[i]) != VK_SUCCESS
352 || vkCreateSemaphore(mDevice.GetVkDevice(), &semaphoreInfo, nullptr, &mRenderFinishedSemaphores[i]) != VK_SUCCESS
353 || vkCreateFence(mDevice.GetVkDevice(), &fenceInfo, nullptr, &mInFlightFences[i]) != VK_SUCCESS) {
354 throw std::runtime_error("Failed to create synchronization objects for a frame!");
355 }
356 }

359 VkSurfaceFormatKHR VRE_SwapChain::ChooseSwapSurfaceFormat(const std::vector<VkSurfaceFormatKHR> &availableFormats)
360 {
361 for (const auto &availableFormat : availableFormats)
362 if (availableFormat.format == VK_FORMAT_B8G8R8A8_SRGB && availableFormat.colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR)
363 return availableFormat;
364

380 VkExtent2D VRE_SwapChain::ChooseSwapExtent(const VkSurfaceCapabilitiesKHR &capabilities)
381 {
382 if (capabilities.currentExtent.width != std::numeric_limits<uint32_t>::max())
383 return capabilities.currentExtent;
384 else {
385 VkExtent2D actualExtent = mWindowExtent;
386 actualExtent.width = std::max(capabilities.minImageExtent.width, std::min(capabilities.maxImageExtent.width, actualExtent.width));
387 actualExtent.height = std::max(capabilities.minImageExtent.height, std::min(capabilities.maxImageExtent.height, actualExtent.height));
388
389 return actualExtent;
390 }

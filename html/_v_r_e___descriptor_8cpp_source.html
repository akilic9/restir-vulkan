49VRE::VRE_DescriptorSetLayout::VRE_DescriptorSetLayout(VRE_Device& device, std::unordered_map<uint32_t, VkDescriptorSetLayoutBinding> bindings)
50 : mDevice(device)
51 , mBindings(bindings)
52{
53 std::vector<VkDescriptorSetLayoutBinding> setLayoutBindings{};
54 for (auto e : mBindings)
55 setLayoutBindings.push_back(e.second);
56
57 VkDescriptorSetLayoutCreateInfo descriptorSetLayoutInfo{};
58 descriptorSetLayoutInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_LAYOUT_CREATE_INFO;
59 descriptorSetLayoutInfo.bindingCount = static_cast<uint32_t>(setLayoutBindings.size());
60 descriptorSetLayoutInfo.pBindings = setLayoutBindings.data();
61
62 if (vkCreateDescriptorSetLayout(mDevice.GetVkDevice(), &descriptorSetLayoutInfo, nullptr, &mDescriptorSetLayout) != VK_SUCCESS)
63 throw std::runtime_error("Failed to create descriptor set layout!");
64}

95bool VRE::VRE_DescriptorPool::AllocateDescriptorSet(const VkDescriptorSetLayout descriptorSetLayout, VkDescriptorSet& descriptor) const
96{
97 VkDescriptorSetAllocateInfo allocInfo{};
98 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
99 allocInfo.descriptorPool = mDescriptorPool;
100 allocInfo.pSetLayouts = &descriptorSetLayout;
101 allocInfo.descriptorSetCount = 1;
102
103 // TODO: Might want to create a "DescriptorPoolManager" class that handles this case, and builds a new pool whenever an old pool fills up.

105 return false;
106
107 return true;

125VRE::VRE_DescriptorWriter& VRE::VRE_DescriptorWriter::WriteBuffer(uint32_t binding, VkDescriptorBufferInfo* bufferInfo)
126{
127 assert(mSetLayout.mBindings.count(binding) == 1 && "Layout does not contain specified binding.");
128
129 auto& bindingDescription = mSetLayout.mBindings[binding];
130
131 assert(bindingDescription.descriptorCount == 1 && "Binding has single descriptor info, but binding expects multiple.");
132
133 VkWriteDescriptorSet write{};
134 write.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
135 write.descriptorType = bindingDescription.descriptorType;
136 write.dstBinding = binding;
137 write.pBufferInfo = bufferInfo;
138 write.descriptorCount = 1;
139
140 mWrites.push_back(write);
141 return *this;
142}

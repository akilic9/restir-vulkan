54void VRE::VRE_Buffer::WriteToBuffer(void* data, VkDeviceSize size, VkDeviceSize offset)
55{
56 assert(mMapped && "Cannot copy to unmapped buffer.");
57
58 if (size == VK_WHOLE_SIZE)
59 memcpy(mMapped, data, mBufferSize);
60 else {
61 char* memOffset = (char*)mMapped;
62 memOffset += offset;
63 memcpy(memOffset, data, size);
64 }
65}

68{

113{

70void VRE::VRE_Buffer::WriteToBuffer(void* data, VkDeviceSize size, VkDeviceSize offset)
71{
72 assert(mMapped && "Cannot copy to unmapped buffer.");
73
74 if (size == VK_WHOLE_SIZE)
75 memcpy(mMapped, data, mBufferSize);
76 else {
77 char* memOffset = (char*)mMapped;
78 memOffset += offset;
79 memcpy(memOffset, data, size);
80 }
81}
